    #!/usr/bin/env python3
"""Render compact Codex quota details for tmux status-right."""

from __future__ import annotations

import json
import os
import shutil
import subprocess
import sys
import time
from datetime import datetime, timezone
from pathlib import Path
from typing import Any


CACHE_TTL_SECONDS = int(os.environ.get("CODEX_STATUSLINE_TTL_SECONDS", "60"))
STALE_TTL_SECONDS = int(os.environ.get("CODEX_STATUSLINE_STALE_SECONDS", "900"))
REQUEST_TIMEOUT_SECONDS = int(os.environ.get("CODEX_STATUSLINE_REQUEST_TIMEOUT_SECONDS", "8"))


def cache_dir() -> Path:
    base = Path(os.environ.get("XDG_CACHE_HOME", Path.home() / ".cache"))
    return base / "codex-statusline"


def cache_path() -> Path:
    return cache_dir() / "tmux.json"


def lock_path() -> Path:
    return cache_dir() / "tmux.lock"


def read_cache() -> dict[str, Any] | None:
    try:
        return json.loads(cache_path().read_text())
    except (FileNotFoundError, json.JSONDecodeError, OSError):
        return None


def write_cache(line: str) -> None:
    cache_dir().mkdir(parents=True, exist_ok=True)
    tmp_path = cache_path().with_suffix(".json.tmp")
    tmp_path.write_text(json.dumps({"line": line, "updated_at": int(time.time())}) + "\n")
    tmp_path.replace(cache_path())


def acquire_lock() -> bool:
    cache_dir().mkdir(parents=True, exist_ok=True)
    try:
        fd = os.open(lock_path(), os.O_CREAT | os.O_EXCL | os.O_WRONLY)
    except FileExistsError:
        return False
    with os.fdopen(fd, "w") as handle:
        handle.write(str(os.getpid()))
    return True


def release_lock() -> None:
    try:
        lock_path().unlink()
    except FileNotFoundError:
        pass


def start_refresh() -> None:
    subprocess.Popen(
        [sys.executable, __file__, "--refresh"],
        stdin=subprocess.DEVNULL,
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
        start_new_session=True,
    )


def parse_timestamp(value: Any) -> datetime | None:
    if value in (None, ""):
        return None
    if isinstance(value, (int, float)):
        return datetime.fromtimestamp(value, tz=timezone.utc).astimezone()
    if isinstance(value, str):
        text = value.replace("Z", "+00:00")
        try:
            return datetime.fromisoformat(text).astimezone()
        except ValueError:
            return None
    return None


def format_reset(value: Any) -> str:
    reset = parse_timestamp(value)
    if reset is None:
        return ""

    now = datetime.now().astimezone()
    if reset.date() == now.date():
        return reset.strftime("@%H:%M")
    if (reset.date() - now.date()).days < 7:
        return reset.strftime("@%a %H:%M")
    return reset.strftime("@%m/%d %H:%M")


def percent_left(window: dict[str, Any]) -> str:
    used = window.get("usedPercent")
    if used is None:
        used = window.get("used_percent")
    if used is None:
        return ""
    try:
        left = max(0, min(100, round(100 - float(used))))
    except (TypeError, ValueError):
        return ""
    return f"{left}% left"


def compact_window(label: str, window: dict[str, Any]) -> str:
    left = percent_left(window)
    reset = format_reset(window.get("resetsAt") or window.get("resets_at"))
    if left and reset:
        return f"{label} {left} {reset}"
    if left:
        return f"{label} {left}"
    if reset:
        return f"{label} {reset}"
    return ""


def first_payload(payload: Any) -> dict[str, Any]:
    if isinstance(payload, list):
        for item in payload:
            if isinstance(item, dict) and item.get("provider") == "codex":
                return item
        for item in payload:
            if isinstance(item, dict):
                return item
        return {}
    if isinstance(payload, dict):
        return payload
    return {}


def render_from_codexbar(payload: Any) -> str:
    data = first_payload(payload)
    usage = data.get("usage") or {}
    identity = usage.get("identity") or {}

    parts = [
        compact_window("5h", usage.get("primary") or {}),
        compact_window("W", usage.get("secondary") or {}),
    ]

    plan = (
        data.get("planType")
        or usage.get("planType")
        or identity.get("planType")
        or identity.get("loginMethod")
    )
    if plan:
        parts.append(str(plan).lower())

    credits = data.get("credits") or usage.get("credits") or {}
    remaining = credits.get("remaining")
    if remaining is None:
        remaining = credits.get("balance")
    if credits.get("unlimited"):
        parts.append("cr unlimited")
    elif remaining not in (None, ""):
        parts.append(f"cr {remaining}")

    rendered = " | ".join(part for part in parts if part)
    return f"Codex {rendered}" if rendered else ""


def fetch_codexbar() -> str:
    if shutil.which("codexbar") is None:
        return ""

    proc = subprocess.run(
        [
            "codexbar",
            "--provider",
            "codex",
            "--source",
            "cli",
            "--format",
            "json",
            "--json-only",
        ],
        check=False,
        stdout=subprocess.PIPE,
        stderr=subprocess.DEVNULL,
        text=True,
        timeout=REQUEST_TIMEOUT_SECONDS,
    )
    if proc.returncode != 0 or not proc.stdout.strip():
        return ""
    return render_from_codexbar(json.loads(proc.stdout))


def rpc_request(proc: subprocess.Popen[str], message: dict[str, Any]) -> None:
    assert proc.stdin is not None
    proc.stdin.write(json.dumps(message, separators=(",", ":")) + "\n")
    proc.stdin.flush()


def fetch_codex_rpc() -> str:
    if shutil.which("codex") is None:
        return ""

    proc = subprocess.Popen(
        ["codex", "-s", "read-only", "-a", "untrusted", "app-server"],
        stdin=subprocess.PIPE,
        stdout=subprocess.PIPE,
        stderr=subprocess.DEVNULL,
        text=True,
    )
    try:
        rpc_request(
            proc,
            {
                "id": 1,
                "method": "initialize",
                "params": {
                    "clientInfo": {
                        "name": "dotfiles-codex-statusline",
                        "version": "0.1.0",
                    }
                },
            },
        )
        rpc_request(proc, {"id": 2, "method": "account/rateLimits/read", "params": {}})
        rpc_request(proc, {"id": 3, "method": "account/read", "params": {}})

        deadline = time.time() + REQUEST_TIMEOUT_SECONDS
        responses: dict[int, dict[str, Any]] = {}
        assert proc.stdout is not None
        while time.time() < deadline and (2 not in responses or 3 not in responses):
            line = proc.stdout.readline()
            if not line:
                break
            message = json.loads(line)
            message_id = message.get("id")
            if message_id in (2, 3):
                responses[message_id] = message
    finally:
        proc.terminate()
        try:
            proc.wait(timeout=1)
        except subprocess.TimeoutExpired:
            proc.kill()

    rate_limits = responses.get(2, {}).get("result", {}).get("rateLimits") or {}
    account = responses.get(3, {}).get("result", {}).get("account") or {}
    usage = {
        "primary": rate_limits.get("primary") or {},
        "secondary": rate_limits.get("secondary") or {},
        "planType": rate_limits.get("planType") or account.get("planType"),
        "credits": rate_limits.get("credits") or {},
    }
    return render_from_codexbar({"usage": usage})


def refresh() -> None:
    if not acquire_lock():
        return
    try:
        try:
            line = fetch_codexbar() or fetch_codex_rpc()
        except Exception:
            line = ""
        write_cache(line)
    finally:
        release_lock()


def main() -> int:
    if len(sys.argv) > 1 and sys.argv[1] == "--refresh":
        refresh()
        return 0

    cached = read_cache()
    now = int(time.time())
    if cached:
        age = now - int(cached.get("updated_at", 0))
        line = str(cached.get("line") or "")
        if age <= CACHE_TTL_SECONDS:
            print(line)
            return 0
        if age <= STALE_TTL_SECONDS:
            start_refresh()
            print(line)
            return 0

    start_refresh()
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
